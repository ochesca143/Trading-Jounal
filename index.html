Google Sheets sync

Every trade you save will automatically sync to a Google Sheet. Your data lives permanently outside the browser — if you ever clear your cache or switch computers, your trades are safe.

One-time setup (3 minutes):

  1. Go to sheets.google.com and create a new spreadsheet
  2. Click Extensions → Apps Script
  3. Delete any existing code, paste the script below, click Save
  4. Click Deploy → New deployment → Web app
  5. Set "Who has access" to Anyone, click Deploy
  6. Copy the Web app URL and paste it below
function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = JSON.parse(e.postData.contents); if (data.action === 'save') { const t = data.trade; const entry = Array.isArray(t.entry) ? t.entry.join(', ') : (t.entry || ''); sheet.appendRow([ t.id, t.date, t.lb, t.inst, t.dir, t.mc, t.setup, entry, t.ss, t.contracts, t.pnl, t.rr, t.exec, t.protected, t.grade, t.notes ]); } if (data.action === 'delete') { const id = data.id; const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 0; i--) { if (String(rows[i][0]) === String(id)) { sheet.deleteRow(i + 1); } } } if (data.action === 'init') { if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Date','Label','Instrument','Direction','Context', 'Setup','Entry Criteria','Stop Size (pts)','Contracts', 'P&L ($)','R','Execution','Protected','Grade','Notes']); } } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
